# Start by installing the Azure CLI using this command in powershell (Administrator)
$ProgressPreference = 'SilentlyContinue'; Invoke-WebRequest -Uri https://aka.ms/installazurecliwindows -OutFile .\AzureCLI.msi; Start-Process msiexec.exe -Wait -ArgumentList '/I AzureCLI.msi /quiet'; Remove-Item .\AzureCLI.msi

# You can verify that it has installed when you restart powershell and then type the following command
az version

# next login using the following command. then follow the instructions you get
az login
